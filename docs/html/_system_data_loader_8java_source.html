60  public void generateAllSystemEmailRules() {
61  // turn off authorization if we're saving
62  context.turnOffAuthorization();
63 
64  // get the Submitted state
65  State submitted = Spring.getBeanOfType(StateManager.class).getState("Submitted");
66  // get all of the current Submitted rules
67  List<EmailWorkflowRule> submittedRules = settingRepo.findEmailWorkflowRulesByState(submitted);
68  // our Always conditions
69  AbstractWorkflowRuleCondition conditionAdvisor = settingRepo.createEmailWorkflowRuleCondition(ConditionType.Always);
70  AbstractWorkflowRuleCondition conditionStudent = settingRepo.createEmailWorkflowRuleCondition(ConditionType.Always);
71  // our email templates to use for the default system email rules
72  EmailTemplate advisorEmailTemplate = settingRepo.findEmailTemplateByName("SYSTEM Advisor Review Request");
73  EmailTemplate studentEmailTemplate = settingRepo.findEmailTemplateByName("SYSTEM Initial Submission");
74  // always send to advisor on submission
75  EmailWorkflowRule advisorRule = settingRepo.createEmailWorkflowRule(submitted);
76  advisorRule.setCondition(conditionAdvisor); // always save a new copy of condition
77  advisorRule.setEmailTemplate(advisorEmailTemplate);
78  advisorRule.setRecipientType(RecipientType.Advisor);
79  advisorRule.setIsSystem(true);
80  // always send to student on submission
81  EmailWorkflowRule studentRule = settingRepo.createEmailWorkflowRule(submitted);
82  studentRule.setCondition(conditionStudent); // always save a new copy of condition
83  studentRule.setEmailTemplate(studentEmailTemplate);
84  studentRule.setRecipientType(RecipientType.Student);
85  studentRule.setIsSystem(true);
86 
87  boolean foundAdvisor = false, foundStudent = false;
88 
89  for (EmailWorkflowRule rule : submittedRules) {
90  // if this is a system-installed rule
91  if (rule.isSystem()) {
92  // compare conditions for advisor rule
93  if (rule.getCondition().getConditionType() == advisorRule.getCondition().getConditionType()) {
94  // compare recipients for advisor rule
95  if (rule.getRecipientType() == advisorRule.getRecipientType()) {
96  foundAdvisor = true;
97  }
98  }
99  // compare conditions for student rule
100  if (rule.getCondition().getConditionType() == studentRule.getCondition().getConditionType()) {
101  // compare recipients for student rule
102  if (rule.getRecipientType() == studentRule.getRecipientType()) {
103  foundStudent = true;
104  }
105  }
106  }
107  }
108  // if the rules weren't found, add them
109  if (!foundAdvisor) {
110  if (advisorRule.getCondition() != null) {
111  advisorRule.getCondition().save();
112  }
113  advisorRule.enable();
114  advisorRule.save();
115  }
116  if (!foundStudent) {
117  if (studentRule.getCondition() != null) {
118  studentRule.getCondition().save();
119  }
120  studentRule.enable();
121  studentRule.save();
122  }
123 
124  // turn on authorization after we're done
125  context.restoreAuthorization();
126  }

149  public EmailTemplate loadSystemEmailTemplate(String name) {
150  try {
151  String templatePath = BASE_PATH + encodeTemplateName(name);
152  File templateFile = new File(templatePath);
153 
154  String data = FileUtils.readFileToString(templateFile);
155 
156  // Remove any comment lines
157  data = data.replaceAll("\\s*#.*[\\n\\r]{1}", "");
158 
159  // Extract the subject
160  Matcher subjectMatcher = SUBJECT_PATTERN.matcher(data);
161  if (!subjectMatcher.find())
162  throw new IllegalStateException("Unable to identify the template's subject.");
163  String subject = subjectMatcher.group(1).trim();
164 
165  // Trim the subject leaving just the body.
166  int index = data.indexOf("\n");
167  if (index < 0)
168  index = data.indexOf("\r");
169  String message = data.substring(index);
170 
171  if (subject == null || subject.length() == 0)
172  throw new IllegalStateException("Unable to identify the template's subject.");
173 
174  if (message == null || message.length() == 0)
175  throw new IllegalStateException("Unable to identify the template's message.");
176 
177  EmailTemplate template = settingRepo.createEmailTemplate(name, subject, message);
178  template.setSystemRequired(true);
179 
180  return template;
181 
182  } catch (Exception e) {
183  throw new IllegalStateException("Unable to generate system email template: " + name, e);
184  }
185

188  public void generateAllSystemEmailTemplates() {
189  // turn off authorization if we're saving
190  context.turnOffAuthorization();
191  for (String name : getAllSystemEmailTemplateNames()) {
192 
193  // try to see if it already exists in the DB
194  EmailTemplate dbTemplate = settingRepo.findSystemEmailTemplateByName(name);
195 
196  // create template or upgrade the old one
197  if (dbTemplate == null) {
198  dbTemplate = loadSystemEmailTemplate(name);
199  Logger.info("New System Email template being installed [%s]", dbTemplate.getName());
200  dbTemplate.save();
201  } else {
202  EmailTemplate loadedTemplate = loadSystemEmailTemplate(name);
203  // if the template in the DB doesn't match in content with the one loaded from .email file
204  if (!(dbTemplate.getMessage().equals(loadedTemplate.getMessage())) || !(dbTemplate.getSubject().equals(loadedTemplate.getSubject()))) {
205  EmailTemplate possibleCustomTemplate = settingRepo.findNonSystemEmailTemplateByName(name);
206  // if this System template already has a custom template (meaning one named the same but that is !isSystemRequired)
207  if (possibleCustomTemplate != null) {
208  // a custom version of this System email template already exists, it's safe to override dbTemplate's data and save
209  dbTemplate.setSystemRequired(false);
210  dbTemplate.setMessage(loadedTemplate.getMessage());
211  dbTemplate.setSubject(loadedTemplate.getSubject());
212  dbTemplate.setSystemRequired(true);
213  Logger.info("Upgrading Old System Email Template for [%s]", dbTemplate.getName());
214  dbTemplate.save();
215  }
216  // there is no custom one yet, we need to make the dbTemplate !isSystemRequired and the save loadedTemplate
217  else {
218  Logger.info("Upgrading Old System Email Template and creating custom version for [%s]", dbTemplate.getName());
219  dbTemplate.setSystemRequired(false);
220  dbTemplate.save();
221  loadedTemplate.save();
222  }
223  }
224  }
225  }
226  // turn on authorization after we're done
227  context.restoreAuthorization();
228  }
