899  private static String validateToken(String token, String type) {
900 
901  String rawToken = null;
902  try {
903  String privateKey = Play.configuration.getProperty("application.secret").substring(0, 16);
904  byte[] raw = privateKey.getBytes();
905  SecretKeySpec skeySpec = new SecretKeySpec(raw, "AES");
906  Cipher cipher = Cipher.getInstance("AES");
907  cipher.init(Cipher.DECRYPT_MODE, skeySpec);
908  rawToken = new String(cipher.doFinal(Base64.decodeBase64(token)));
909  } catch (Exception e) {
910  // Any error should just return null.
911  return null;
912  }
913 
914  // Check that we were able to decrypt something.
915  if (rawToken == null)
916  return null;
917 
918  // Check that we have the three parts of the rawToken.
919  String[] parts = rawToken.split(":");
920  if (parts.length != 3)
921  return null;
922 
923  // Check that the token was generated today or yesterday.
924  Date now = new Date();
925  int date = now.getYear() + now.getMonth() + now.getDate();
926  if (!(String.valueOf(date).equals(parts[0]) ||
927  String.valueOf(date-1).equals(parts[0])))
928  return null;
929 
930  // Check that the token is of the expected type.
931  if (!type.equals(parts[2]))
932  return null;
933 
934  // All checks passed, return the email address.
935  return parts[1];
936  }
